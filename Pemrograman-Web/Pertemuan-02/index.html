ğŸ› ï¸ Langkah-Langkah Praktikum

a. Buat File index.js


// index.js
const express = require('express');
const cors = require('cors');
const app = express();
const PORT = 3001;

app.use(cors());
app.use(express.json());

app.get('/', (req, res) => {
  res.send('Hello World from Express.js!');
});

app.post('/data', (req, res) => {
  const { nama } = req.body;
  res.send(\`Data diterima: \${nama}\`);
});

app.listen(PORT, () => {
  console.log(\`Server berjalan di http://localhost:\${PORT}\`);
});

b. Jalankan Server


node index.js

Uji di browser: http://localhost:3001

c. Commit & Push


git add .
git commit -m "Add basic Express.js setup and routes"
git push origin main

d. Test GET & POST di Postman

  • GET: http://localhost:3001/
  • POST: http://localhost:3001/data (JSON body)
