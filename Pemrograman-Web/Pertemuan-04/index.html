      function DataFetcher() {
      // Mendeklarasikan state data dengan nilai awal berupa array kosong
        const [data, setData] = useState([]);
         
         // useEffect dijalankan sekali saat komponen pertama kali dimuat (mount)
         useEffect(() => {
           // Melakukan pengambilan data dari API
           fetch('https://jsonplaceholder.typicode.com/posts')
             .then((res) => res.json()) // Mengubah response menjadi format JSON
             .then((result) => setData(result)); // Menyimpan data yang diambil ke dalam state 'data'
         }, []); // Array kosong memastikan efek hanya berjalan sekali (saat mount)
          
         return (
